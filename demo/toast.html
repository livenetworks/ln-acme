Toast

Нотификации со страничен accent и иконка. Автоматски исчезнуваат после 6 секунди.

Примери

Error со листа на грешки

HTML & JS API

<!-- Toast container (еднаш на страница) -->
<div data-ln-toast></div>

<!-- JS API -->
<script>
// Основен toast
window.lnToast.enqueue({
    type: 'success',     // success | error | warn | info
    title: 'Наслов',
    message: 'Опис...'
});

// Error со листа
window.lnToast.enqueue({
    type: 'error',
    title: 'Валидација',
    message: ['Грешка 1', 'Грешка 2', 'Грешка 3']
});

// Исчисти сите
window.lnToast.clear();
</script>

<!-- Статичен toast (хидриран при init) -->
<div data-ln-toast>
    <div data-ln-toast-item data-type="success" data-title="OK">
        Зачувано!
    </div>
</div>
