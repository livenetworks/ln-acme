    ln-progress — Progress Bar

    65% (130/200)
    55% (11/20) — red
    60% (12/20) — green
    75% (15/20) — yellow

    Box HTML

    <div data-ln-box>
    <header>
        Box Title
        <button data-ln-box-action="collapse">Collapse</button>
        <button data-ln-box-action="expand">Expand</button>
    </header>
    <main data-ln-box-body>
        <!-- содржина -->
    </main>
</div>

<!-- Collapsed по default -->
<div data-ln-box="collapsed">...</div>

    Progress HTML

    <div data-ln-progress>
    <div data-ln-progress="65"
         data-ln-progress-max="100"
         class="green">
    </div>
</div>

<!-- Бои: class="green" | "red" | "yellow" -->
<!-- max default = 100 (може да се испушти) -->

<!-- Динамичка промена -->
<script>
    el.setAttribute('data-ln-progress', '80');
    // MutationObserver автоматски ќе ја ажурира ширината
</script>
