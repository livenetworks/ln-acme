    File upload со drag-drop, progress bar, валидација на тип. Потребен е сервер за вистински upload.

    Upload Zone (демо — без сервер)

    Довлечи фајлови тука или кликни за избор

    Дозволени: pdf, doc, docx, jpg, png

      HTML

      <!-- Upload container -->
<div data-ln-upload="/api/upload-endpoint"
     data-ln-upload-accept="pdf,doc,jpg,png">

    <!-- Drop zone -->
    <div class="ln-upload__zone">
        <p>Довлечи фајлови тука</p>
    </div>

    <!-- File list (автоматски се полни) -->
    <ul class="ln-upload__list"></ul>

</div>

<!-- Атрибути -->
data-ln-upload="URL"           - upload endpoint
data-ln-upload-accept="ext,ext" - дозволени екстензии

<!-- JS API (достапно на container елементот) -->
<script>
const container = document.querySelector('[data-ln-upload]');
container.lnUploadAPI.getFileIds();   // [1, 2, 3]
container.lnUploadAPI.getFiles();     // file objects
container.lnUploadAPI.clear();        // исчисти сè
</script>

<!-- Custom Events -->
ln-upload:uploaded   - {id, name, size}
ln-upload:removed    - {id}
ln-upload:error      - {message}
ln-upload:invalid    - {file, reason}

<!-- Hidden inputs за форма -->
<!-- Автоматски се креираат: -->
<input type="hidden" name="file_ids[]" value="123">
