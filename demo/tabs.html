Events

Lifecycle настани

НастанКогаdetail
ln-tabs:changeПо активирање на нов таб{ key, tab, panel }

Живо демо

Уште нема настани — кликни таб.

Код

document.getElementById('user-tabs').addEventListener('ln-tabs:change', function(e) {
    console.log('Активен таб:', e.detail.key);
});

// Глобално — за сите tab секции
document.addEventListener('ln-tabs:change', function(e) {
    analytics.track('tab_change', { tab: e.detail.key });
});

HTML

<!-- Namespace = id атрибут (или data-ln-tabs-key) -->
<section id="my-tabs" data-ln-tabs data-ln-tabs-default="first">

    <!-- Tab копчиња -->
    <nav>
        <button data-ln-tab="first">Прв таб</button>
        <button data-ln-tab="second">Втор таб</button>
    </nav>

    <!-- Панели -->
    <section data-ln-panel="first">
        Содржина на прв таб
    </section>

    <section data-ln-panel="second" class="hidden">
        Содржина на втор таб
    </section>

</section>

<!-- Атрибути -->
id="key"                   - namespace за hash (задолжително за повеќе секции)
data-ln-tabs-key="key"     - алтернативно, ако id не е погоден
data-ln-tabs               - контејнер
data-ln-tabs-default="x"   - почетен таб
data-ln-tabs-focus         - фокусирај прв input при промена
data-ln-tab="key"          - таб копче
data-ln-panel="key"        - панел содржина

<!-- Hash формат: #my-tabs:second → активирај second таб -->
<!-- Повеќе секции: #my-tabs:second&other-tabs:info -->
<!-- Без id/data-ln-tabs-key → нема hash sync, само defaultKey -->
