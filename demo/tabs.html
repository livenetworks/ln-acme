    HTML

    <div data-ln-tabs data-ln-tabs-default="first">

    <!-- Tab копчиња -->
    <nav>
        <button data-ln-tab="first">Прв таб</button>
        <button data-ln-tab="second">Втор таб</button>
    </nav>

    <!-- Панели -->
    <div data-ln-panel="first">
        Содржина на прв таб
    </div>

    <div data-ln-panel="second" class="hidden">
        Содржина на втор таб
    </div>

</div>

<!-- Атрибути -->
data-ln-tabs              - контејнер
data-ln-tabs-default="x"  - почетен таб
data-ln-tabs-focus         - фокусирај прв input при промена
data-ln-tab="key"          - таб копче
data-ln-panel="key"        - панел содржина

<!-- Hash sync: #second → активирај second таб -->
